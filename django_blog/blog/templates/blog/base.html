from django.db.models import Q from django.shortcuts import render from .models import Post def search(request): query = request.GET.get('q') results = Post.objects.filter( Q(title__icontains=query) | Q(content__icontains=query) | Q(tags__name__icontains=query) ).distinct() return render(request, 'blog/search_results.html', {'posts': results, 'query': query})
